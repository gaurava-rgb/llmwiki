from snowflake.snowpark.context import get_active_session

import streamlit as st

session = get_active_session()

st.title("The Engagement Gap")

st.markdown("Understanding team engagement patterns and identifying top performers")

df = session.sql("SELECT * FROM HACKATHON.DATA.ENGAGEMENT_LOGS").to_pandas()

st.sidebar.header("Filters")

org_units = ['All'] + sorted(df['ORG_UNIT'].unique().tolist())

selected_org = st.sidebar.selectbox("Organization Unit", org_units)

filtered_df = df if selected_org == 'All' else df[df['ORG_UNIT'] == selected_org]

col1, col2, col3, col4 = st.columns(4)

st.metric("Total Users", filtered_df['USER_ID'].nunique())

st.metric("Total Teams", filtered_df['TEAM'].nunique())

st.metric("Avg Engagement", f"{filtered_df['ENGAGEMENT_SCORE'].mean():.2f}")

st.metric("Total Sessions", f"{filtered_df['SESSION_COUNT'].sum():,}")

st.markdown("---")

st.subheader("Team Rankings by Average Engagement Score")

team_stats = filtered_df.groupby('TEAM').agg({

'ENGAGEMENT_SCORE': 'mean',

'SESSION_COUNT': 'sum',

}).reset_index()

team_stats.columns = ['Team', 'Avg Engagement Score', 'Total Sessions', 'Unique Users']

team_stats = team_stats.sort_values('Avg Engagement Score', ascending=False)

st.bar_chart(team_stats.set_index('Team')['Avg Engagement Score'])

st.dataframe(team_stats.style.format({

'Avg Engagement Score': '{:.2f}',

}), use_container_width=True)

st.markdown("---")

st.subheader("Engagement Trends: Top vs Bottom Team")

top_team = team_stats.iloc[0]['Team']

bottom_team = team_stats.iloc[-1]['Team']

trend_data = filtered_df[filtered_df['TEAM'].isin([top_team, bottom_team])]

trend_pivot = trend_data.groupby(['WEEK_START', 'TEAM'])['ENGAGEMENT_SCORE'].mean().reset_index()

trend_pivot = trend_pivot.pivot(index='WEEK_START', columns='TEAM', values='ENGAGEMENT_SCORE')

st.line_chart(trend_pivot)

st.caption(f"Top Team: {top_team} (Avg: {team_stats.iloc[0]['Avg Engagement Score']:.2f}) | Bottom Team: {bottom_team} (Avg:

{team_stats.iloc[-1]['Avg Engagement Score']:.2f})")

st.markdown("---")

st.subheader("Top 10 Users by Total Sessions")

user_leaderboard = filtered_df.groupby('USER_ID').agg({

'SESSION_COUNT': 'sum',

'ENGAGEMENT_SCORE': 'mean',

}).reset_index()

user_leaderboard.columns = ['User ID', 'Total Sessions', 'Avg Engagement', 'Team']

user_leaderboard = user_leaderboard.sort_values('Total Sessions', ascending=False).head(10)

st.dataframe(user_leaderboard.style.format({

'Total Sessions': '{:,.0f}',

}).background_gradient(subset=['Total Sessions'], cmap='Greens'), use_container_width=True)

st.markdown("---")

st.subheader("Bonus: Engagement by Category")

category_stats = filtered_df.groupby('CATEGORY')['ENGAGEMENT_SCORE'].mean().sort_values(ascending=False)

st.bar_chart(category_stats)

col1, col2 = st.columns(2)

st.metric("Highest Engagement Category", category_stats.index[0], f"{category_stats.iloc[0]:.2f}")

st.metric("Lowest Engagement Category", category_stats.index[-1], f"{category_stats.iloc[-1]:.2f}")
